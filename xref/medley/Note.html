13  /**
14   * {@code Note} class that contains all the informations about a musical note.
15   * This class represents a single note. A note is identified by a frequency and
16   * by a MIDI note number.
17   *
18   * <p>A note has also a {@code Name}, represented using English literal notation
19   * an {@code Accidental}, that represent the alteration of a note, and an
20   * octave.
21   *
22   * <p>A {@code Note} can be easily transposed using the methods
23   * {@link #semitoneUp() semitoneUp}, {@link #semitoneDown() semitoneDown} and
24   * {@link #transpose(int semitones) transpose}.
25   *
26   * @author Salvatore Gentile
27   * @see medley.score.Element
28   * @see medley.score.ScoreElement
29   */
30  public class Note extends ScoreElement {

75       * frequency value, {@code Duration} and volume value. The new {@code Note}
76       * will be created using the value of the nearest valid note frequency.
77       *
78       * @param frequency a specified frequency value.
79       * @param duration a specified {@code Duration}.
80       * @param volume a specified volume value.
81       * @throws IllegalArgumentException
82       *         if the specified frequency is not a positive double between 8.176
83       *         and 12543.854.
84       * @see medley.util.Frequency#getMIDINumber(double frequency)
85       * @see medley.tone.Tone#Tone(double frequency)
86       */
87      public Note(double frequency, Duration duration, int volume)
88              throws IllegalArgumentException {
89          super(duration, volume);
90  
91          this.midiNumber = Frequency.getMIDINumber(frequency);
92          this.frequency = frequency;
93          this.tone = new Tone(frequency);
94          this.octave = (midiNumber / 12) - 1;
95      }

242      * @param octave a specified octave.

247     public Note(Tone tone, int octave, Duration duration, int volume) {
248         super(duration, volume);
249 
250         this.tone = tone;
251         this.octave = octave;
252         this.midiNumber = tone.getPitchClass() + 12 * (octave + 1);
253         this.frequency = Frequency.getMIDINumberFrequency(midiNumber);
254     }

339     /**
340      * Transposes this {@code Note} by a specified number of half steps.
341      *
342      * <p>This operation does not affect the {@code Duration} and the volume of
343      * the note.
344      *
345      * @param semitones a specified number of half steps.
346      * @see medley.util.Validator#transpose(int midiNumber, int semitones)
347      */
348     public void transpose(int semitones) throws IllegalArgumentException {
349         if (semitones == 0) {
350             return;
351         }
352 
353         Validator.transpose(midiNumber, semitones);
354 
355         this.midiNumber += semitones;
356         this.frequency = Frequency.getMIDINumberFrequency(midiNumber);
357         this.tone = new Tone(midiNumber);
358         this.octave = (midiNumber / 12) - 1;
359     }

361     /**
362      * Switches the {@code Accidental} of this {@code Note} with an equivalent
363      * one.
364      */
365     public void switchAccidental() {
366         // Adjusts the octave when switching from C to B-Sharp...
367         if (tone.getName() == Name.C
368                 && tone.getAccidental() == Accidental.Natural) {
369             octave--;
370         }
371 
372         // ...and viceversa.
373         if (tone.getName() == Name.B
374                 && tone.getAccidental() == Accidental.Sharp) {
375             octave++;
376         }
377 
378         tone.switchAccidental();
379     }

387     public String toStringExt() {
388         StringBuilder sb = new StringBuilder();
389 
390         sb.append("\nNOTE:\t\t");
391         sb.append(this.toString());
392 
393         sb.append("\nName:\t\t");
394         sb.append(tone.getName());
395 
396         sb.append("\nAccidental:\t");
397         sb.append(tone.getAccidental().getName());
398 
399         sb.append("\nOctave:\t\t");
400         sb.append(octave);
401 
402         sb.append("\nFrequency:\t");
403         sb.append(String.format(Locale.ENGLISH, "%1.3f", frequency));
404         sb.append("Hz");
405 
406         sb.append("\nMIDI number:\t");
407         sb.append(midiNumber);
408 
409         sb.append("\nDuration:\t");

411 
412         sb.append("\nDuration Value:\t");
413         sb.append(duration.getDurationValue());

415         sb.append("\nVolume:\t\t");
416         sb.append(volume);
417

423     /**
424      * Returns a string representation of this {@code Note}. The string consists
425      * of a concatenation of the name, an eventual accidental's symbol and the
426      * octave of the note.
427      *
428      * @return a string representation of this {@code Note}.
429      */
430     @Override
431     public String toString() {
432         StringBuilder sb = new StringBuilder(3);
433 
434         sb.append(tone.getName());
435 
436         if (tone.getAccidental() != Accidental.Natural) {
437             sb.append(tone.getAccidental());
438         }
439 
440         sb.append(octave);
441 
442         return sb.toString();
443     }

445     /**
446      * Compares the specified object with this {@code Note} for equality.
447      *
448      * <p>This implementation first checks if the specified object is a
449      * {@code Note}. If not, it returns {@code false}; if so, it checks if the
450      * specified object is this {@code Note}. If not, it returns {@code false};
451      * if so, it compares the frequency of both notes.
452      *
453      * @param o object to be compared for equality with this {@code Note}.
454      * @return {@code true} if the specified object is equals to this
455      *         {@code Note}; {@code false} otherwise.
456      */
457     @Override
458     public boolean equals(Object o) {
459         if (!(o instanceof Note)) {
460             return false;
461         }
462 
463         if (o == this) {
464             return true;
465         }
466 
467         Note n = (Note) o;
468 
469         if (n.getFrequency() == this.frequency
470                 && n.duration.equals(this.duration)) {
471             return true;
472         }
473 
474         return false;
475     }

477     /**
478      * Returns the hash code value for this {@code Note}.
479      *
480      * @return the hash code value for this {@code Note}.
481      */
482     @Override
483     public int hashCode() {
484         final int prime = 911;
485 
486         int hash = super.hashCode();
487         hash = prime * hash + new Double(frequency).hashCode();
488         hash = prime * hash + midiNumber;
489         hash = prime * hash + tone.hashCode();
490         hash = prime * hash + octave;
491 
492         return hash;
493     }
