3   import medley.score.Element;

6   /**
7    * {@code Duration} class that contains all the informations about musical
8    * duration and time span.
9    *
10   * @author Salvatore Gentile
11   */
12  public class Duration {
13  
14      /**
15       * Musical values, that represent the notes durations.
16       */
17      public enum Value {
18          /**
19           * Breve has the time value of two semibreves.
20           */
21          Breve(2),
22  
23          /**
24           * Semibreve has the time value of half a {@code Breve} or the time
25           * value of two minims or the time value of four crotchets.
26           */
27          Semibreve(1),
28  
29          /**
30           * Minim has the time value of half a {@code Semibreve} or the time
31           * value of two crotchets.
32           */
33          Minim(1 / 2.0),
34  
35          /**
36           * Crotchet has the time value of half a {@code Minim} or the time value
37           * of a quarter of a {@code Semibreve}.
38           */
39          Crotchet(1 / 4.0),
40  
41          /**
42           * Quaver has the time value of half a {@code Crotchet} or the time
43           * value of a quarter of a {@code Minim} or the time value of an eighth
44           * of a {@code Semibreve}.
45           */
46          Quaver(1 / 8.0),
47  
48          /**
49           * Semiquaver has the time value of half a {@code Quaver}.
50           */
51          Semiquaver(1 / 16.0),
52  
53          /**
54           * Demisemiquaver has the time value of half a {@code Semiquaver}.
55           */
56          Demisemiquaver(1 / 32.0),
57  
58          /**
59           * Hemidemisemiquaver has the time value of half a
60           * {@code Demisemiquaver}.
61           */
62          Hemidemisemiquaver(1 / 64.0);
63  
64          private final double value;
65  
66          Value(double value) {
67              this.value = value;
68          }
69  
70          /**
71           * Returns the duration value of this {@code Value}.
72           *
73           * @return the duration value of this {@code Value}.
74           */
75          public double getDurationValue() {
76              return value;
77          }
78      }
79  
80      private Value value;
81      private int dots;
82  
83      /**
84       * Initializes a newly created {@code Duration} object with the specified
85       * {@code Value}.
86       *
87       * @param value a specified {@code Duration} value.
88       */
89      public Duration(Value value) {
90          this.value = value;
91          this.dots = Element.DEFAULT_DOTS;
92      }
93  
94      /**
95       * Initializes a newly created {@code Duration} object with the
96       * specified {@code Value} and dots number.
97       *
98       * @param value a specified {@code Value} duration.
99       * @param dots a specified number of dots.
100      * @throws IllegalArgumentException
101      *         if the specified dots number is not a positive integer between 0
102      *         and 3.
103      */
104     public Duration(Value value, int dots) throws IllegalArgumentException {
105         Validator.dots(dots);
106 
107         this.value = value;
108         this.dots = dots;
109     }
110 
111     /**
112      * Initializes a newly created {@code Duration} object with the same
113      * {@code Value} and dots number of the specified note. In other words, the
114      * newly created duration is a copy of the argument duration.
115      *
116      * @param duration a {@code Duration}.
117      */
118     public Duration(Duration duration) {
119         this.value = duration.value;
120         this.dots = duration.dots;
121     }
122 
123     /**
124      * Returns the {@code Value} of this {@code Duration}.
125      *
126      * @return the {@code Value} of this {@code Duration}.
127      */
128     public Value getValue() {
129         return value;
130     }
131 
132     /**
133      * Replaces the {@code Value} of this {@code Duration} with the specified
134      * one.
135      *
136      * @param value a specified {@code Value}.
137      */
138     public void setValue(Value value) {
139         this.value = value;
140     }
141 
142     /**
143      * Returns the number of dots of this {@code Duration}.
144      *
145      * @return the number of dots of this {@code Duration}.
146      */
147     public int getDots() {
148         return dots;
149     }
150 
151     /**
152      * Replace the number of dots of this {@code Duration} with the
153      * specified one.
154      *
155      * @param dots a specified number of dots.
156      * @throws IllegalArgumentException
157      *         if the specified dots number is not a positive integer between 0
158      *         and 3.
159      * @see medley.util.Validator#dots(int dots)
160      */
161     public void setDots(int dots) throws IllegalArgumentException {
162         Validator.dots(dots);
163 
164         this.dots = dots;
165     }
166 
167     /**
168      * Returns the duration value of this {@code Duration}.
169      *
170      * @return the duration value of this {@code Duration}.
171      */
172     public double getDurationValue() {
173         if (dots != 0) {
174             switch (dots) {
175                 case 1:
176                     return increaseDurationValue(1 / 2.0);
177 
178                 case 2:
179                     return increaseDurationValue(1 / 2.0 + 1 / 4.0);
180 
181                 case 3:
182                     return increaseDurationValue(1 / 2.0 + 1 / 4.0 + 1 / 8.0);
183             }
184         }
185 
186         return value.getDurationValue();
187     }
188 
189     private double increaseDurationValue(double increment) {
190         return value.getDurationValue() + value.getDurationValue() * increment;
191     }
192 
193     /**
194      * Returns a verbose string representation of this {@code Duration}. The
195      * string contains all the informations about the duration.
196      *
197      * @return a verbose string representation of this {@code Duration}.
198      */
199     public String verboseString() {
200         return null;
201     }
202 
203     /**
204      * Returns a string representation of this {@code Duration}. The string
205      * consists of a concatenation of the value and eventual dots.
206      *
207      * @return a string representation of this {@code Duration}.
208      */
209     @Override
210     public String toString() {
211         StringBuilder sb = new StringBuilder();
212 
213         sb.append(value);
214 
215         if (dots > 0) {
216             sb.append(" with ");
217             sb.append(dots);
218             sb.append((dots == 1 ? " dot" : " dots"));
219         }
220 
221         return sb.toString();
222     }
223 
224     /**
225      * Compares the specified object with this {@code Duration} for equality.
226      *
227      * @param o object to be compared for equality with this {@code Duration}.
228      * @return {@code true} if the specified object is equals to this
229      *         {@code Duration}; {@code false} otherwise.
230      */
231     @Override
232     public boolean equals(Object o) {
233         if (!(o instanceof Duration)) {
234             return false;
235         }
236 
237         if (o == this) {
238             return true;
239         }
240 
241         Duration d = (Duration) o;
242 
243         if (d.getDurationValue() != this.getDurationValue()) {
244             return false;
245         }
246 
247         if (d.getValue() == this.value && d.getDots() == this.dots){
248             return true;
249         }
250 
251         return false;
252     }
253 
254     /**
255      * Returns the hash code value for this {@code Duration}.
256      *
257      * @return the hash code value for this {@code Duration}.
258      */
259     @Override
260     public int hashCode() {
261         final int prime = 911;
262 
263         int hash = 1;
264         hash = prime * hash + value.hashCode();
265         hash = prime * hash + dots;
266 
267         return hash;
268     }
